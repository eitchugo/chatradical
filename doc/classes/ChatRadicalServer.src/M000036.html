    def ClienteMotd()
        motd = String.new()
        begin
            File.open('motd.txt', 'r') do |arquivo|
                while linha = arquivo.gets
                    motd = motd + linha
                end
            end
            return motd
        rescue
            puts "error> Erro ao ler o arquivo MOTD! Mostrando mensagem meia-boca."
            return "Bem vindo ao ChatRadical!"
        end
    end
