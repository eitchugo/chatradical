    def CliVerLog(cliente, sala, tail)
        if tail == nil
            tail = 30
        else
            tail = tail.to_i
        end

        cliente.puts tail

        log = @salas[sala].VerLog(tail)

        log.each do |linha_log|
            cliente.puts "RCV_CHATLOG [LOG] #{linha_log}"
        end
        return 0
    end
