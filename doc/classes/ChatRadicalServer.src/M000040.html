    def NovaSala(nome, descricao=nil)
        unless @salas[nome]
            if descricao == nil
                return false
            end

            @salas[nome] = Sala.new(nome, descricao)
            puts "server> Nova sala criada: #{nome} - #{descricao}"

            return true
        end
        return true
    end
