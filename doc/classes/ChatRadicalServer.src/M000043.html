    def ChatNotice(usuario, msg)
        msg = Time.now.strftime("[%H:%M:%S] ") + "*** #{msg}"

        sala = @salas[usuario.sala]
        sala.AdicionarLog(msg)

        membros = sala.membros
        membros.each do |cliente|
            cliente.puts "RCV_CHATMSG #{msg}"
        end

        return "RCV_CHAT OK"
    end
